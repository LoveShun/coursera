\begin{equation}
\begin{aligned}
	h_{\theta}(x) & = \theta^{T}X = X^T\theta \\
	& = 
		\left(
			\begin{matrix}
				1 & x_1 & x_2 & \dots & x_n
			\end{matrix}
		\right)
		\left(
			\begin{matrix}
			\theta_0 \\
			\theta_1 \\
			\dots \\
			\theta_n
			\end{matrix}
		\right) \\
	& = \theta_0 + \theta_1x_1 + \theta_2x_2 + \dots + \theta_nx_n
\end{aligned}
\end{equation}
